Password (.vbs)

@echo off
cls
:password
set input=
set/p input=password (Input then press Enter)
if %input%==(Put Password Here) goto Yes
if not %input%==(Put Password Here) goto No


:Yes
(Put Batch File or Website Here) ((ex:Start c:\(file location))

No:
echo Incorrect Password
goto password
